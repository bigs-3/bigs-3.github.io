1.6. Statistical

  • friends count
  • group members count
query {
  relation_aggregate(where: {is_friend: {_eq: true}, _and: {from: {_eq: "0x0a0b364093cb37787827e210806f50c30ce4e192"}, _or: {to: {_eq: "0x0a0b364093cb37787827e210806f50c30ce4e192"}}}}) {
    aggregate {
      count(columns: is_friend)
    }
  }
}
query MyQuery {
  group_member_aggregate {
    aggregate {
      count(columns: member)
    }
    nodes {
      member
    }
  }
}
